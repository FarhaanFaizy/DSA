#include<stdio.h>
int main(){
    int c[3][3] = {{0,4,11},{6,0,2},{3,999,0}},n=3,A[3][3];
    for (int i=0;i<n;i++)
        for(int j=0;j<n;j++)
                    A[i][j]=c[i][j];
    for(int k=0;k<n;k++){
        for(int i=0;i<n;i++)
            for(int j=0;j<n;j++){
                int min=A[i][j],x=A[i][k]+A[k][j];
                if(x<min)
                    min=x;
                A[i][j]=min;
            }
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++)
            printf("%d ",A[i][j]);
        printf("\n");
    }
}
