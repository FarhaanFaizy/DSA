#include <stdio.h>
void display(int *a, int n)
{
    int i;
    for (i=0;i<n;i++)
    {
        printf("%d ",a[i]);
    }
}
int main()
{
    int n,i,j,k=0;
    int *a,b[100];
    printf("\nEnter number of elements:");
    scanf("%d",&n);
    a=(int*)malloc(n*sizeof(int));
    printf("\nEnter all the elements:");
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    printf("\nElements in array are ");
    display(a,n);
    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
            if (a[i]==a[j])
            {
                b[k]=a[i];
                k++;
                break;
            }
        }
    }
    printf("\nRepeated elements are ");
    display(b,k);
}
